## Animations generated in R version 3.4.1 (2017-06-30) using the
##   package animation
library(animation)
library(ROCR)
library(gplots)
for (i in 1:100) {
    curve(sin(x), from = -5 + (i * 0.05), to = 5 + (i * 0.05), 
        col = "red", ylab = "")
    curve(cos(x), from = -5 + (i * 0.05), to = 5 + (i * 0.05), 
        add = TRUE, col = "blue", ylab = "")
    legend("topright", legend = c("sin(x)", "cos(x)"), fill = c("red", 
        "blue"), bty = "n")
}
